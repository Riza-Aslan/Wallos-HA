#include <tunables/global>

profile wallos flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>
  #include <abstractions/openssl>

  capability chown,
  capability dac_override,
  capability fowner,
  capability fsetid,
  capability kill,
  capability mknod,
  capability net_bind_service,
  capability setgid,
  capability setuid,
  capability sys_chroot,

  signal (receive) peer=wallos,

  network inet tcp,
  network inet udp,
  network inet6 tcp,
  network inet6 udp,

  /usr/local/bin/php ix,
  /usr/sbin/nginx ix,
  /usr/sbin/crond ix,
  /usr/bin/dos2unix ix,

  /var/www/html/** r,
  /var/www/html/db/** rw,
  /var/www/html/images/uploads/** rw,
  /var/www/html/.tmp/** rw,

  /etc/nginx/** r,
  /etc/php/** r,
  /etc/cron.d/** r,
  /etc/localtime r,
  /etc/timezone r,
  /usr/share/zoneinfo/** r,

  /var/log/** w,
  /tmp/** rw,
  /run/** rw,

  /proc/sys/kernel/random/uuid r,
  /proc/sys/kernel/osrelease r,
}
